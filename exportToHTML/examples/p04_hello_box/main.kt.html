58   class SimpleClock( 
59       key: Any?, 
60       val format: JsTimeFormat, 
61       val locale: String = "en-us" 
62   ) : Widget<JsDate>(key) { 
63     override val initialState: JsDate 
64       get() = JsDate() 
65    
66     var timeId: Int? = null 
67    
68     override fun init() { 
69       timeId = setInterval(::updateClock, 1000) 
70       console.log("SimpleClock: timer created") 
71     } 
72    
73     fun updateClock() { 
74       updateState { JsDate() } 
75     } 
76    
77     override fun render(state: JsDate) = Row { 
78       +state.toLocaleTimeString(locale, format) 
79     } 
80    
81     override fun dispose() { 
82       clearInterval(timeId!!) 
83       console.log("SimpleClock: timer destroyed") 
84     } 
85   }
